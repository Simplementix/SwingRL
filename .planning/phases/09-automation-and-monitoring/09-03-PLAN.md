SQLite tables in db/trading_ops.db:
```sql
-- portfolio_snapshots
CREATE TABLE portfolio_snapshots (
    timestamp TEXT NOT NULL,
    environment TEXT NOT NULL,
    total_value REAL NOT NULL,
    equity_value REAL, crypto_value REAL, cash_balance REAL,
    high_water_mark REAL, daily_pnl REAL, drawdown_pct REAL,
    PRIMARY KEY (timestamp, environment)
);

-- trade_log
CREATE TABLE trade_log (
    trade_id TEXT PRIMARY KEY,
    timestamp TEXT NOT NULL,
    environment TEXT NOT NULL,
    symbol TEXT NOT NULL,
    side TEXT NOT NULL,
    quantity REAL NOT NULL,
    fill_price REAL NOT NULL,
    commission REAL,
    slippage REAL,
    broker TEXT
);

-- circuit_breaker_events
CREATE TABLE circuit_breaker_events (
    event_id TEXT PRIMARY KEY,
    timestamp TEXT NOT NULL,
    environment TEXT NOT NULL,
    trigger_type TEXT NOT NULL,
    trigger_value REAL,
    threshold REAL,
    action TEXT,
    cooldown_end TEXT
);

-- risk_decisions
CREATE TABLE risk_decisions (
    decision_id TEXT PRIMARY KEY,
    timestamp TEXT NOT NULL,
    environment TEXT NOT NULL,
    symbol TEXT, proposed_action TEXT, final_action TEXT,
    risk_rule_triggered TEXT, reason TEXT
);
```

    **dashboard/Dockerfile.dashboard:**
    ```dockerfile
    FROM python:3.11-slim

    # Non-root user matching bot container convention
    RUN useradd --create-home --uid 1000 trader
    WORKDIR /app

    # Install dashboard deps only (no torch, no SB3)
    COPY dashboard/requirements.txt /app/requirements.txt
    RUN pip install --no-cache-dir -r requirements.txt

    # Copy dashboard code
    COPY dashboard/ /app/dashboard/

    USER trader

    # Streamlit config: disable telemetry, headless mode
    ENV STREAMLIT_BROWSER_GATHER_USAGE_STATS=false
    ENV STREAMLIT_SERVER_HEADLESS=true
    ENV STREAMLIT_SERVER_PORT=8501

    EXPOSE 8501

    HEALTHCHECK --interval=30s --timeout=5s --retries=3 \
        CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8501/_stcore/health')"

    CMD ["python", "-m", "streamlit", "run", "dashboard/app.py", "--server.address=0.0.0.0"]
    ```
