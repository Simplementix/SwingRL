    ```python
    """MacroWatcher: FRED API-based economic event calendar.

    Checks for upcoming high-impact macro releases (FOMC, CPI, NFP, etc.)
    using FRED's fred/release/dates endpoint. Caches release schedules
    weekly to minimize API calls.
    """
    from __future__ import annotations

    from datetime import UTC, datetime, timedelta
    from typing import TYPE_CHECKING, Any

    import structlog

    if TYPE_CHECKING:
        from swingrl.config.schema import MacroWatcherConfig

    log = structlog.get_logger(__name__)
    ```

    Class `MacroWatcher`:
    - `__init__(self, config: MacroWatcherConfig)`:
      - `self._config = config`
      - `self._cache: dict[str, list[datetime]] = {}` — event_name -> list of release datetimes
      - `self._cache_expiry: datetime | None = None`
      - `self._fred_api_key = os.environ.get("FRED_API_KEY", "")`

    - `get_nearest_event(self) -> tuple[str | None, float | None]`:
      - Refresh cache if expired or empty
      - Find the nearest future event across all release IDs
      - If nearest is within `lookforward_hours`, return `(event_name, hours_until)`
      - Otherwise return `(None, None)`

    - `_refresh_cache(self) -> None`:
      - For each `(event_name, release_id)` in `config.release_ids`:
        - Call FRED API: `https://api.stlouisfed.org/fred/release/dates?release_id={id}&api_key={key}&file_type=json&include_release_dates_with_no_data=true`
        - Parse response JSON, extract `release_dates[].date` as datetime objects
        - Filter to future dates only
        - Store in `self._cache[event_name]`
      - Set `self._cache_expiry = datetime.now(UTC) + timedelta(days=config.cache_days)`
      - Wrap entire refresh in try/except — on any error, log warning and leave cache as-is

    - `_fetch_release_dates(self, release_id: int) -> list[datetime]`:
      - Use fredapi library: `from fredapi import Fred; fred = Fred(api_key=self._fred_api_key)`
      - `fred.get_release_dates(release_id)` returns dates
      - Or use urllib directly if fredapi doesn't have this method — check fredapi docs
      - Wrap in try/except, return empty list on error

    **Step 2 — Create `scripts/show_schedule.py`:**

    ```python
    """Print all APScheduler job next fire times in ET and UTC.

    Usage: uv run python scripts/show_schedule.py
    """
    from __future__ import annotations

    import datetime
    from zoneinfo import ZoneInfo

    from swingrl.config.schema import load_config

    ET = ZoneInfo("America/New_York")
    UTC_TZ = datetime.timezone.utc


    def main() -> None:
        config = load_config("config/swingrl.yaml")
        from apscheduler.schedulers.background import BackgroundScheduler

        scheduler = BackgroundScheduler()

        # Import and register jobs — same as main.py
        from scripts.main import create_scheduler_and_register_jobs
        create_scheduler_and_register_jobs(scheduler, config)

        print(f"{'Job ID':<30} {'Next ET':<28} {'Next UTC':<28}")
        print("-" * 86)
        for job in scheduler.get_jobs():
            next_utc = job.trigger.get_next_fire_time(
                None, datetime.datetime.now(UTC_TZ)
            )
            if next_utc:
                next_et = next_utc.astimezone(ET)
                print(f"{job.id:<30} {str(next_et):<28} {str(next_utc):<28}")
            else:
                print(f"{job.id:<30} {'N/A':<28} {'N/A':<28}")


    if __name__ == "__main__":
        main()
    ```

    **Step 1 — Create `scripts/deploy.sh`:**

    ```bash
    #!/usr/bin/env bash
    # SwingRL production deployment script.
    # Usage: bash scripts/deploy.sh [--smoke-test]
    #
    # Steps:
    #   1. git pull (update to latest)
    #   2. docker compose build (rebuild images)
    #   3. docker compose down (stop old containers)
    #   4. Pre-create bind mount dirs with correct ownership
    #   5. docker compose up -d (start new containers)
    #   6. Wait for all containers healthy (timeout 120s)
    #   7. Optional: run smoke test
    #
    # .env is NEVER touched — operator manages manually.

    set -euo pipefail

    COMPOSE_FILE="docker-compose.prod.yml"
    SMOKE_TEST=false

    if [[ "${1:-}" == "--smoke-test" ]]; then
        SMOKE_TEST=true
    fi

    echo "=== SwingRL Deploy ==="

    # Step 1: Git pull
    echo "[1/6] Pulling latest code..."
    git pull

    # Step 2: Build
    echo "[2/6] Building Docker images (--no-cache)..."
    docker compose -f "$COMPOSE_FILE" build --no-cache

    # Step 3: Stop
    echo "[3/6] Stopping existing containers..."
    docker compose -f "$COMPOSE_FILE" down

    # Step 4: Pre-create dirs
    echo "[4/6] Ensuring bind mount directories..."
    mkdir -p data db models logs status config
    # Set ownership for trader user (UID 1000) if running as root
    if [ "$(id -u)" -eq 0 ]; then
        chown 1000:1000 data db models logs status
    fi

    # Step 5: Start
    echo "[5/6] Starting containers..."
    docker compose -f "$COMPOSE_FILE" up -d

    # Step 6: Wait healthy
    echo "[6/6] Waiting for containers to become healthy (timeout 120s)..."
    TIMEOUT=120
    ELAPSED=0
    while [ $ELAPSED -lt $TIMEOUT ]; do
        HEALTHY=$(docker compose -f "$COMPOSE_FILE" ps --format json 2>/dev/null | python3 -c "
    import sys, json
    healthy = 0
    total = 0
    for line in sys.stdin:
        svc = json.loads(line)
        total += 1
        if 'healthy' in svc.get('Health', '') or svc.get('State', '') == 'running':
            healthy += 1
    print(f'{healthy}/{total}')
    " 2>/dev/null || echo "0/0")
        echo "  Health: $HEALTHY (${ELAPSED}s)"
        if [[ "$HEALTHY" == "3/3" ]]; then
            echo "All containers healthy!"
            break
        fi
        sleep 5
        ELAPSED=$((ELAPSED + 5))
    done

    if [ $ELAPSED -ge $TIMEOUT ]; then
        echo "WARNING: Timeout waiting for healthy status. Check logs."
        docker compose -f "$COMPOSE_FILE" logs --tail=20
    fi

    # Optional smoke test
    if [ "$SMOKE_TEST" = true ]; then
        echo "=== Running smoke test ==="
        docker exec swingrl python scripts/smoke_test.py 2>&1 || echo "Smoke test failed!"
    fi

    echo "=== Deploy complete ==="
    ```
