    ```python
    """Live memory agent client with fail-open semantics.

    Queries the 5 /live/ endpoints on the memory service for cycle-level
    trading decisions. Every method returns a safe default on any error
    so that trading is never blocked by memory agent unavailability.
    """
    from __future__ import annotations

    import json
    import urllib.error
    import urllib.request
    from typing import Any

    import structlog

    log = structlog.get_logger(__name__)


    class LiveMemoryClient:
        """Fail-open HTTP client for memory agent live trading endpoints."""

        def __init__(
            self,
            base_url: str = "http://swingrl-memory:8889",
            timeout: float = 3.0,
            api_key: str = "",
        ) -> None:
            self._base_url = base_url.rstrip("/")
            self._timeout = timeout
            self._api_key = api_key

        def _post(self, path: str, payload: dict[str, Any]) -> dict[str, Any]:
            """POST JSON to memory service endpoint. Returns empty dict on any failure."""
            url = f"{self._base_url}{path}"
            try:
                data = json.dumps(payload).encode("utf-8")
                headers: dict[str, str] = {"Content-Type": "application/json"}
                if self._api_key:
                    headers["X-API-Key"] = self._api_key
                req = urllib.request.Request(url, data=data, headers=headers, method="POST")
                with urllib.request.urlopen(req, timeout=self._timeout) as resp:  # nosec B310
                    return json.loads(resp.read().decode("utf-8"))
            except Exception as exc:
                log.warning("live_memory_request_failed", url=url, error=str(exc))
                return {}
    ```

    **5 public methods** (one per endpoint):

    `query_cycle_gate(env: str, context: dict) -> bool`:
    - POST to `/live/cycle_gate` with `{"env": env, "context": context}`
    - On success: return `result.get("approved", True)` — but if confidence < 0.3, return True (approve)
    - On failure: return True (fail-open — allow cycle)

    `query_blend_weights(env: str, context: dict) -> dict[str, float]`:
    - POST to `/live/blend_weights` with `{"env": env, "context": context}`
    - On success with confidence >= 0.3: return `result.get("deltas", {})`
    - On failure or low confidence: return `{}` (no weight adjustment)

    `query_risk_thresholds(env: str, context: dict) -> tuple[float | None, float | None]`:
    - POST to `/live/risk_thresholds`
    - On success with confidence >= 0.3: return `(result.get("max_dd"), result.get("daily_limit"))`
    - On failure: return `(None, None)` (no tightening)

    `query_position_advice(env: str, context: dict) -> dict[str, Any]`:
    - POST to `/live/position_advice`
    - On success with confidence >= 0.3: return parsed response
    - On failure: return `{"kelly_scale_factor": 1.0, "hold_bias": False, "max_simultaneous_positions": None}`

    `query_trade_veto(env: str, context: dict, signal: dict) -> bool`:
    - POST to `/live/trade_veto` with `{"env": env, "context": context, "signal": signal}`
    - On success with confidence >= 0.3: return `result.get("vetoed", False)`
    - On failure or low confidence: return False (don't veto)

    `ingest_cycle_summary(env: str, summary: dict) -> bool`:
    - POST to memory service `/ingest` with `{"text": json.dumps(summary), "source": f"live:{env}:cycle_summary"}`
    - Fire-and-forget, return True/False
