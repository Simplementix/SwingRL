2. Add `_check_bootstrap_guard` function BEFORE `evaluate_shadow_promotion`:
   ```python
   def _check_bootstrap_guard(
       config: Any,
       db: Any,
       env_name: str,
   ) -> tuple[bool, str]:
       """Check dual conditions: min live trades AND min days since shadow deploy.

       Uses LIVE trades table (not shadow_trades) -- bootstrap guards against
       spurious promotion before enough real market performance data exists.

       Args:
           config: SwingRLConfig with retraining settings.
           db: DatabaseManager instance.
           env_name: Environment name (equity or crypto).

       Returns:
           Tuple of (passed, reason). reason is empty string if passed.
       """
       # Get min thresholds from config (per-environment)
       if env_name == "equity":
           min_trades = config.retraining.bootstrap_min_trades_equity
           min_days = config.retraining.bootstrap_min_days_equity
       else:
           min_trades = config.retraining.bootstrap_min_trades_crypto
           min_days = config.retraining.bootstrap_min_days_crypto

       # Count LIVE trades (not shadow_trades)
       with db.sqlite() as conn:
           row = conn.execute(
               "SELECT COUNT(*) as cnt FROM trades WHERE environment = ?",
               (env_name,),
           ).fetchone()
           trade_count = row["cnt"] if row else 0

       if trade_count < min_trades:
           return False, f"bootstrap: {trade_count}/{min_trades} trades"

       # Check days since shadow deployment
       with db.sqlite() as conn:
           row2 = conn.execute(
               "SELECT shadow_deployed_at FROM retrain_state WHERE environment = ?",
               (env_name,),
           ).fetchone()

       if row2 is None or row2["shadow_deployed_at"] is None:
           return False, "bootstrap: no shadow deployment recorded"

       deployed = datetime.fromisoformat(row2["shadow_deployed_at"])
       if deployed.tzinfo is None:
           deployed = deployed.replace(tzinfo=UTC)
       days_elapsed = (datetime.now(tz=UTC) - deployed).days

       if days_elapsed < min_days:
           return False, f"bootstrap: {days_elapsed}/{min_days} days elapsed"

       return True, ""
   ```
