1. **Add `_check_bootstrap_guard` helper to src/swingrl/shadow/promoter.py** (before `evaluate_shadow_promotion`):

   ```python
   def _check_bootstrap_guard(
       config: Any,
       db: Any,
       env_name: str,
   ) -> tuple[bool, str]:
       """Check dual bootstrap conditions before shadow promotion.

       Conditions (BOTH must be met):
       1. Minimum live trades (from trades table, NOT shadow_trades)
       2. Minimum calendar days since shadow deployment

       Uses config.retraining.bootstrap_min_trades_{env} and
       config.retraining.bootstrap_min_days_{env} thresholds.

       Args:
           config: SwingRLConfig with retraining settings.
           db: DatabaseManager instance.
           env_name: Environment name (equity or crypto).

       Returns:
           Tuple of (passed: bool, reason: str). Empty reason if passed.
       """
       min_trades = getattr(config.retraining, f"bootstrap_min_trades_{env_name}", 50)
       min_days = getattr(config.retraining, f"bootstrap_min_days_{env_name}", 14)

       # Count LIVE trades (NOT shadow_trades — Pitfall 2 from RESEARCH.md)
       with db.sqlite() as conn:
           row = conn.execute(
               "SELECT COUNT(*) as cnt FROM trades WHERE environment = ?",
               (env_name,),
           ).fetchone()
           trade_count = row["cnt"] if row else 0

       if trade_count < min_trades:
           return False, f"bootstrap: {trade_count}/{min_trades} trades"

       # Check days since shadow deployment from retrain_state
       with db.sqlite() as conn:
           row2 = conn.execute(
               "SELECT shadow_deployed_at FROM retrain_state WHERE environment = ?",
               (env_name,),
           ).fetchone()

       if row2 is None or row2["shadow_deployed_at"] is None:
           return False, "bootstrap: no shadow deployment recorded"

       from datetime import UTC, datetime  # noqa: PLC0415

       deployed = datetime.fromisoformat(row2["shadow_deployed_at"])
       if deployed.tzinfo is None:
           deployed = deployed.replace(tzinfo=UTC)
       days_elapsed = (datetime.now(tz=UTC) - deployed).days

       if days_elapsed < min_days:
           return False, f"bootstrap: {days_elapsed}/{min_days} days elapsed"

       return True, ""
   ```
