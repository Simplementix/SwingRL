   ```python
   def _run_retrain_job(env_name: str, trigger_reason: str) -> None:
       """Internal retrain job handler. Acquires lock, spawns subprocess, updates state.

       Called by equity_retrain_job and crypto_retrain_job. Also called by
       performance_check_job when Sharpe degradation triggers retrain.
       """
       ctx = _get_ctx()

       if is_halted(ctx.config, ctx.db):
           log.info("retrain_skipped_halted", env=env_name)
           return

       # Check enabled in retrain_state
       with ctx.db.sqlite() as conn:
           row = conn.execute(
               "SELECT enabled FROM retrain_state WHERE environment = ?",
               (env_name,),
           ).fetchone()
           if row and not row["enabled"]:
               log.info("retrain_skipped_disabled", env=env_name)
               return

       # Acquire sequential lock (non-blocking)
       lock = try_acquire_retrain_lock()
       if lock is None:
           log.warning("retrain_skipped_lock_held", env=env_name, trigger=trigger_reason)
           return

       try:
           log.info("retrain_starting", env=env_name, trigger=trigger_reason)

           # Fire Discord "retrain started" notification
           _is_perf_trigger = trigger_reason.startswith("performance_degradation")
           _level = "warning" if _is_perf_trigger else "info"
           ctx.alerter.send_alert(
               _level,
               f"Retrain Started: {env_name.title()}",
               f"Trigger: {trigger_reason}\nEnvironment: {env_name}",
           )

           # Spawn retrain subprocess
           exit_code = spawn_retrain_subprocess(
               env_name=env_name,
               config_path=Path("config/swingrl.yaml"),
               dry_run=ctx.config.retraining.dry_run,
               scheduler_mode=True,
           )

           # Create orchestrator for state updates
           from swingrl.memory.client import MemoryClient  # noqa: PLC0415
           memory_client = MemoryClient(ctx.config) if ctx.config.memory_agent.api_key else None
           orchestrator = RetrainingOrchestrator(
               config=ctx.config,
               db=ctx.db,
               lifecycle=None,  # Lifecycle handled inside subprocess
               alerter=ctx.alerter,
               memory_client=memory_client,
           )

           # Build minimal result from exit code for failure state tracking
           from swingrl.training.retraining import RetrainResult  # noqa: PLC0415
           status_map = {0: "success", 1: "gate_failed", 2: "crash", 3: "timeout", 4: "data_stale"}
           result = RetrainResult(
               status=status_map.get(exit_code, "crash"),
               exit_code=exit_code,
               env_name=env_name,
           )
           orchestrator.update_failure_state(env_name, result)

           # Fire completion/failure Discord notification
           if exit_code == 0:
               ctx.alerter.send_alert(
                   "info",
                   f"Retrain Completed: {env_name.title()}",
                   f"Trigger: {trigger_reason}\nResult: success\nDeployed to shadow.",
               )
           else:
               ctx.alerter.send_alert(
                   "warning" if exit_code == 1 else "critical",
                   f"Retrain {'Gate Failed' if exit_code == 1 else 'Failed'}: {env_name.title()}",
                   f"Trigger: {trigger_reason}\nExit code: {exit_code}\nStatus: {result.status}",
               )

           log.info("retrain_completed", env=env_name, exit_code=exit_code, status=result.status)

       except Exception:
           log.exception("retrain_job_error", env=env_name)
           ctx.alerter.send_alert(
               "critical",
               f"Retrain Job Error: {env_name.title()}",
               f"Unexpected error in retrain job wrapper. Check logs.",
           )
       finally:
           lock.release()


   def equity_retrain_job() -> None:
       """Monthly equity retraining job. Fired by APScheduler CronTrigger."""
       _run_retrain_job("equity", "scheduled_monthly")


   def crypto_retrain_job() -> None:
       """Biweekly crypto retraining job. Fired by APScheduler CronTrigger."""
       _run_retrain_job("crypto", "scheduled_biweekly")


   def performance_check_job() -> None:
       """Daily check for Sharpe degradation — triggers retrain if thresholds breached.

       For each environment, computes rolling 30-day Sharpe and compares against:
       1. Deployment baseline Sharpe (decline > sharpe_decline_threshold)
       2. Absolute floor (sharpe_floor)

       Either breach triggers an automated retrain for that environment.
       """
       ctx = _get_ctx()

       if is_halted(ctx.config, ctx.db):
           return

       rc = ctx.config.retraining
       for env_name in ("equity", "crypto"):
           current_sharpe = compute_rolling_sharpe(
               ctx.db, env_name,
               window_days=rc.rolling_sharpe_window_days,
               min_observations=rc.rolling_sharpe_min_observations,
           )
           if current_sharpe is None:
               log.info("performance_check_insufficient_data", env=env_name)
               continue

           # Read baseline from retrain_state
           with ctx.db.sqlite() as conn:
               row = conn.execute(
                   "SELECT baseline_sharpe FROM retrain_state WHERE environment = ?",
                   (env_name,),
               ).fetchone()
               baseline = row["baseline_sharpe"] if row and row["baseline_sharpe"] else None

           triggered = False
           reason_parts = []

           # Check absolute floor
           if current_sharpe < rc.sharpe_floor:
               triggered = True
               reason_parts.append(f"Sharpe {current_sharpe:.2f} < floor {rc.sharpe_floor}")

           # Check decline from baseline
           if baseline is not None and baseline > 0:
               decline_pct = (baseline - current_sharpe) / baseline
               if decline_pct > rc.sharpe_decline_threshold:
                   triggered = True
                   reason_parts.append(
                       f"Sharpe declined {decline_pct:.0%} from baseline {baseline:.2f} to {current_sharpe:.2f}"
                   )

           if triggered:
               trigger_reason = f"performance_degradation ({'; '.join(reason_parts)})"
               log.warning("performance_degradation_detected", env=env_name, reason=trigger_reason)
               _run_retrain_job(env_name, trigger_reason)
           else:
               log.info("performance_check_ok", env=env_name, sharpe=round(current_sharpe, 4))
   ```

   ```python
       hour=config.retraining.equity_cron_hour,

       misfire_grace_time=config.scheduler.misfire_grace_time,
   )

   # Crypto retrain: biweekly on Sunday at 3 AM ET
   # Using IntervalTrigger for predictable biweekly (CronTrigger week='*/2' anchoring is ambiguous)
   from datetime import datetime as _dt, timedelta as _td
   import zoneinfo as _zi
   _et = _zi.ZoneInfo("America/New_York")
   _now = _dt.now(_et)
   # Find next Sunday at crypto_cron_hour ET
   _days_until_sunday = (6 - _now.weekday()) % 7 or 7
   _next_sunday = (_now + _td(days=_days_until_sunday)).replace(
       hour=config.retraining.crypto_cron_hour, minute=0, second=0, microsecond=0
   )
   scheduler.add_job(
       crypto_retrain_job,
       trigger="interval",
       weeks=2,
       start_date=_next_sunday,
       timezone="America/New_York",
       id="crypto_retrain",
       replace_existing=True,
       misfire_grace_time=config.scheduler.misfire_grace_time,
   )

   # Performance check: daily at 8 PM ET (after shadow promotion check at 7 PM)
   ```
