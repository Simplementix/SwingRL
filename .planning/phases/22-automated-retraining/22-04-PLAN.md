   def equity_retrain_job() -> None:
       """Monthly equity retraining via subprocess with file lock guard.

       Acquires file lock to prevent concurrent retrains. Spawns retrain
       subprocess with nice +10 priority. Handles exit codes and fires
       Discord alerts. Never raises (wraps in try/except).
       """
       ctx = _get_ctx()

       if is_halted(ctx.db):
           log.warning("equity_retrain_skipped", reason="halt_flag_active")
           return

       lock = filelock.FileLock(str(_RETRAIN_LOCK_PATH))
       try:
           lock.acquire(timeout=0)
       except filelock.Timeout:
           log.warning("retrain_already_running", env="equity")
           return

       try:
           # Fire retrain-started Discord alert
           ctx.alerter.send_alert(
               "info",
               "Retrain Started: Equity",
               "Monthly equity retraining triggered. Estimated duration from last run.",
           )

           from swingrl.training.retraining import spawn_retrain_subprocess  # noqa: PLC0415

           config_path = str(Path("config/swingrl.yaml"))
           exit_code = spawn_retrain_subprocess(
               env_name="equity",
               config_path=config_path,
               trigger_reason="scheduled_monthly",
               dry_run=ctx.config.retraining.dry_run,
           )

           _handle_retrain_exit_code(ctx, "equity", exit_code, "scheduled_monthly")
       except Exception:
           log.exception("equity_retrain_job_failed")
           try:
               ctx.alerter.send_alert("critical", "Equity Retrain Crashed", "Unhandled exception in equity retrain job")
           except Exception:
               log.exception("equity_retrain_alert_failed")
       finally:
           lock.release()
   ```

   def performance_trigger_check_job() -> None:
       """Check rolling Sharpe for performance-triggered retraining (daily).

       For each environment, compute rolling Sharpe and compare against
       deployment baseline. If degradation detected, trigger immediate retrain.
       """
       ctx = _get_ctx()

       if is_halted(ctx.db):
           log.warning("performance_trigger_check_skipped", reason="halt_flag_active")
           return

       try:
           from swingrl.training.retraining import (  # noqa: PLC0415
               check_performance_trigger,
               spawn_retrain_subprocess,
           )

           for env_name in ("equity", "crypto"):
               triggered, reason = check_performance_trigger(ctx.db, ctx.config, env_name)
               if triggered:
                   log.warning("performance_retrain_triggered", env=env_name, reason=reason)
                   ctx.alerter.send_alert(
                       "warning",
                       f"Sharpe Degradation: {env_name.title()}",
                       f"{reason}. Automated retrain triggered.",
                   )

                   lock = filelock.FileLock(str(_RETRAIN_LOCK_PATH))
                   try:
                       lock.acquire(timeout=0)
                   except filelock.Timeout:
                       log.warning("retrain_already_running", env=env_name, trigger="performance")
                       continue

                   try:
                       config_path = str(Path("config/swingrl.yaml"))
                       exit_code = spawn_retrain_subprocess(
                           env_name=env_name,
                           config_path=config_path,
                           trigger_reason=reason,
                       )
                       _handle_retrain_exit_code(ctx, env_name, exit_code, reason)
                   finally:
                       lock.release()
       except Exception:
           log.exception("performance_trigger_check_failed")
   ```

6. Add `_handle_retrain_exit_code(ctx, env_name, exit_code, trigger_reason)`:
   ```python
   def _handle_retrain_exit_code(
       ctx: JobContext, env_name: str, exit_code: int, trigger_reason: str,
   ) -> None:
       """Process retrain subprocess exit code and fire appropriate Discord alert."""
       if exit_code == 0:
           ctx.alerter.send_alert(
               "info",
               f"Retrain Completed: {env_name.title()}",
               f"Retraining succeeded. Trigger: {trigger_reason}. Model deployed to shadow.",
           )
           log.info("retrain_completed", env=env_name, trigger=trigger_reason)
       elif exit_code == 1:
           ctx.alerter.send_alert(
               "warning",
               f"Retrain Gate Failed: {env_name.title()}",
               f"Training completed but ensemble gate failed. Trigger: {trigger_reason}. No shadow deployment.",
           )
           log.warning("retrain_gate_failed", env=env_name, trigger=trigger_reason)
       elif exit_code == 2:
           ctx.alerter.send_alert(
               "critical",
               f"Retrain Crashed: {env_name.title()}",
               f"Training subprocess crashed. Trigger: {trigger_reason}. Check logs.",
           )
           log.error("retrain_crashed", env=env_name, exit_code=exit_code, trigger=trigger_reason)
       elif exit_code == 3:
           ctx.alerter.send_alert(
               "critical",
               f"Retrain Timeout: {env_name.title()}",
               f"Training exceeded {ctx.config.retraining.timeout_hours}h timeout. Trigger: {trigger_reason}.",
           )
           log.error("retrain_timeout", env=env_name, trigger=trigger_reason)
       elif exit_code == 4:
           ctx.alerter.send_alert(
               "warning",
               f"Retrain Stale Data: {env_name.title()}",
               f"Data freshness check failed. Trigger: {trigger_reason}. Retrain aborted.",
           )
           log.warning("retrain_stale_data", env=env_name, trigger=trigger_reason)
       elif exit_code == 5:
           log.info("retrain_disabled", env=env_name)
       else:
           log.error("retrain_unknown_exit_code", env=env_name, exit_code=exit_code)
   ```
