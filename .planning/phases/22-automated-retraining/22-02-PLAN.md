1. **Create src/swingrl/training/rolling_sharpe.py**:
   ```python
   """Rolling Sharpe ratio from live trades for performance-triggered retraining."""
   from __future__ import annotations
   from datetime import UTC, datetime, timedelta
   from typing import TYPE_CHECKING, Any
   import math
   import structlog

   if TYPE_CHECKING:
       from swingrl.data.db import DatabaseManager

   log = structlog.get_logger(__name__)

   # Annualization factors
   _ANNUALIZE = {"equity": math.sqrt(252), "crypto": math.sqrt(1460)}

   def compute_rolling_sharpe(
       db: Any,
       env_name: str,
       window_days: int = 30,
       min_observations: int = 10,
   ) -> float | None:
       """Compute rolling Sharpe from live trades over last window_days.

       Aggregates fills to daily P&L series. Annualizes with sqrt(252) for equity,
       sqrt(1460) for crypto (6 bars/day x 365.25). Returns None if fewer than
       min_observations non-zero P&L days exist in window.
       """
       cutoff = (datetime.now(tz=UTC) - timedelta(days=window_days)).isoformat()
       with db.sqlite() as conn:
           rows = conn.execute(
               "SELECT DATE(timestamp) as trade_date, "
               "SUM(CASE WHEN side='sell' THEN price*quantity ELSE -price*quantity END) as daily_pnl "
               "FROM trades WHERE environment = ? AND timestamp >= ? "
               "GROUP BY DATE(timestamp) ORDER BY trade_date",
               (env_name, cutoff),
           ).fetchall()

       # Filter to non-zero P&L days
       pnl_values = [r["daily_pnl"] for r in rows if r["daily_pnl"] and abs(r["daily_pnl"]) > 1e-9]
       if len(pnl_values) < min_observations:
           log.info("rolling_sharpe_insufficient_data", env=env_name, days=len(pnl_values), min=min_observations)
           return None

       mean_pnl = sum(pnl_values) / len(pnl_values)
       variance = sum((p - mean_pnl) ** 2 for p in pnl_values) / (len(pnl_values) - 1)
       std_pnl = math.sqrt(variance) if variance > 0 else 0.0
       if std_pnl == 0:
           return 0.0

       daily_sharpe = mean_pnl / std_pnl
       annualize = _ANNUALIZE.get(env_name, math.sqrt(252))
       return float(daily_sharpe * annualize)
   ```

**Arguments:**
- `--env` (required for --manual/--enable/--disable): `equity`, `crypto`, or `all`
- `--manual`: trigger immediate retrain with confirmation prompt. Show: last retrain date, current Sharpe, days since last retrain. Ask "Proceed? [y/N]". In --scheduler-mode, skip confirmation.
- `--dry-run`: full pipeline without deploy_to_shadow. Can combine with --manual.
- `--status`: comprehensive status display. For each env: last retrain (date, result, duration), next scheduled (from APScheduler), current model (deployed date, WF Sharpe), shadow (status + bootstrap progress X/50 trades Y/14 days), schedule enabled/disabled, last 5 retrain Sharpe trend with directional arrow (up/down/flat).
- `--enable` / `--disable`: set retrain_state.enabled for the specified env. Reset consecutive_failures on enable.
- `--history`: last 10 retrains default. `--limit N` overrides. Table format default, `--json` for scripting.
- `--scheduler-mode`: internal flag used by APScheduler jobs — skip interactive confirmation, use structured logging only.
- `--timesteps N`: override total_timesteps for smoke testing (e.g., --timesteps 1000).
- `--config PATH`: config file path (default config/swingrl.yaml).

**Entry point logic:**
1. Parse args
2. Load config via `load_config(args.config)`
3. Initialize DatabaseManager, init_schema()
4. Route to appropriate handler:
   - `--status` -> `_show_status(db, config)`
   - `--history` -> `_show_history(db, args.env, args.limit, args.json)`
   - `--enable` -> `_set_enabled(db, args.env, True)`
   - `--disable` -> `_set_enabled(db, args.env, False)`
   - `--manual` -> `_run_manual(db, config, args)`
   - default (no action flag) -> print help
